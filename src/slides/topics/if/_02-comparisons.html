Comparison Operators

Compare numbers, lengths, or match tokens.

Numeric comparisons

    
    /* Equality */
    if(var(--count) = 0, 'empty', 'has items')

    /* Greater than */
    if(var(--speed) > 88, green, orange)

    /* Greater or equal */
    if(var(--level) >= 100, 'max', 'growing')

    /* Less than */
    if(var(--battery) < 20, red, inherit)

    /* Less or equal */
    if(var(--temp) <= 0, blue, inherit)

Token matching

    
    /* Match exact token */
    background: if(
      attr(data-theme token) = dark,
      #1a1a2e,
      #ffffff
    );

    /* Match status */
    border-color: if(
      attr(data-status token) = error,
      red,
      transparent
    );

    /* Match mode */
    animation: if(
      attr(data-mode token) = active,
      pulse 1s infinite,
      none
    );
